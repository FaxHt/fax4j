24   *  <li>From email address would be used as sender email

83       * This function update the fax job from the request data.<br>

88       * @param   faxJob

91      @Override

93      {
94          String targetAddress=null;
95          String senderEmail=null;
96          try
97          {
98              //get target address
99              targetAddress=this.getTargetAddress(inputData);
100             
101             //get sender email
102             senderEmail=this.getSenderEmail(inputData);
103         }
104         catch(MessagingException exception)
105         {
106             throw new FaxException("Unable to extract fax job data from mail message.",exception);
107         }
108         
109         //update fax job
110         faxJob.setTargetAddress(targetAddress);
111         faxJob.setSenderEmail(senderEmail);
112     }

114     /**
115      * Returns the target address from the provided mail object. 
116      * 
117      * @param   mailMessage
118      *          The mail message with the fax data
119      * @return  The target address
120      * @throws  MessagingException
121      *          Any exception while handling the mail message
122      */
123     protected String getTargetAddress(Message mailMessage) throws MessagingException
124     {
125         //by default the target address is taken from the mail subject which
126         //is expected to be in the format of: fax:<number>
127         String subject=mailMessage.getSubject();
128         String targetAddress=null;
129         if((subject!=null)&&(subject.startsWith("fax:"))&&(subject.length()>4))
130         {
131             targetAddress=subject.substring(4);
132         }
133 
134         return targetAddress;
135     }

137     /**
138      * Returns the sender email from the provided mail object. 
139      * 
140      * @param   mailMessage
141      *          The mail message with the fax data
142      * @return  The sender email
143      * @throws  MessagingException
144      *          Any exception while handling the mail message
145      */
146     protected String getSenderEmail(Message mailMessage) throws MessagingException
147     {
148         Address[] addresses=mailMessage.getFrom();
149         String senderEmail=null;
150         if((addresses!=null)&&(addresses.length>0))
151         {
152             //get sender mail address (only first from is used)
153             Address address=addresses[0];
154             
155             //get as string
156             senderEmail=address.toString();
157         }
158 
159         return senderEmail;
160     }

162     /**
163      * Returns the file info from the provided mail object.<br>
164      * This function does not handle multiple file attachments.
165      * 
166      * @param   mailMessage
167      *          The mail message with the fax data
168      * @return  The file info
169      * @throws  MessagingException
170      *          Any exception while handling the mail message
171      * @throws  IOException
172      *          Any IO exception
173      */
174     protected FileInfo getFileInfo(Message mailMessage) throws MessagingException,IOException
175     {
176         //get content
177         Object messageContent=mailMessage.getContent();
178         
179         String charset=null;
180         if(mailMessage instanceof MimeMessage)
181         {
182             //get mime message
183             MimeMessage mimeMessage=(MimeMessage)mailMessage;
184             
185             //get charset 
186             charset=mimeMessage.getEncoding();
187         }
188         
189         String name="fax.txt";
190         byte[] content=null;
191         if(messageContent instanceof String)
192         {
193             //get content
194             content=IOHelper.convertStringToBinary((String)messageContent,charset);
195         }
196         else if(messageContent instanceof Multipart)
197         {
198             //get multi part
199             Multipart multiPart=(Multipart)messageContent;
200             
201             //get parts amount
202             int amount=multiPart.getCount();
203             
204             Part part=null;
205             DataHandler dataHandler=null;
206             if(amount>0)
207             {
208                 //handle only the first body part
209                 part=multiPart.getBodyPart(0);
210                 
211                 //get data handler
212                 dataHandler=part.getDataHandler();
213                 
214                 //get values
215                 name=dataHandler.getName();
216                 if(name==null)
217                 {
218                     name="fax.txt";
219                 }
220 
221                 //get data
222                 content=IOHelper.readStream(part.getInputStream());
223             }
224         }
225         
226         //create file info
227         FileInfo fileInfo=new FileInfo(name,content);
228 
229         return fileInfo;
230     }
