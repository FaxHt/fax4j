15  public class ApacheCommonsLogger extends AbstractLogger

18      protected final Log APACHE_LOGGER;

20      /**
21       * This is the default constructor.
22       */
23      public ApacheCommonsLogger()
24      {
25          super();
26  
27          //init logger
28          this.APACHE_LOGGER=ApacheCommonsLogger.initializeLogger();
29      }
30      
31      /**
32       * This function initializes and returns the logger.
33       * 
34       * @return  The logger
35       */
36      protected static final Log initializeLogger()
37      {
38          //get logger
39          Log logger=LogFactory.getLog(FaxClient.class.getName());
40  
41          return logger;
42      }
43  
44      /**
45       * Logs the provided data.
46       *  
47       * @param   level
48       *          The log level
49       * @param   message
50       *          The message parts (may be null)
51       * @param   throwable
52       *          The error (may be null)
53       */
54      @Override
55      protected void logImpl(LogLevel level,Object[] message,Throwable throwable)
56      {
57          //format log message (without exception)
58          String text=this.formatLogMessage(level,message,null);
59  
60          //get log level
61          int levelValue=level.getValue();
62          switch(levelValue)
63          {
64              case LogLevel.DEBUG_LOG_LEVEL_VALUE:
65                  this.APACHE_LOGGER.debug(text);
66                  break;
67              case LogLevel.ERROR_LOG_LEVEL_VALUE:
68                  this.APACHE_LOGGER.error(text,throwable);
69                  break;
70              case LogLevel.INFO_LOG_LEVEL_VALUE:
71              default:
72                  this.APACHE_LOGGER.info(text);
73                  break;
74          }
75      }
