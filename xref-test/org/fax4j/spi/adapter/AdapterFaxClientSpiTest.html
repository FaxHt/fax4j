265     @Test
266     public void ensureAllMappedSPIsAreValidatedAndHaveConditionsConfigurationTest() throws Exception
267     {
268         Map<String,String> properties=this.emptyFaxClientSpi.getAllConfiguration();
269         Iterator<?> iterator=properties.entrySet().iterator();
270         String typeString=this.emptyFaxClientSpi.getConfigurationValue("org.fax4j.spi.adapter.internal.spi.types");
271         Assert.assertNotNull(typeString);
272         Assert.assertNotSame(Integer.valueOf(0),Integer.valueOf(typeString.length()));
273         while(iterator.hasNext())
274         {
275             Entry<?,?> entry=(Entry<?,?>)iterator.next();
276             String key=(String)entry.getKey();
277             
278             if(key.startsWith("org.fax4j.spi.type.map."))
279             {
280                 String type=key.substring("org.fax4j.spi.type.map.".length());
281                 
282                 if((!type.equals("test"))&&(!type.equals("adapter")))
283                 {
284                     String value=(String)entry.getValue();
285                     Assert.assertNotNull(value);
286                     Assert.assertNotSame("No value found for property: "+key,Integer.valueOf(0),Integer.valueOf(value.length()));
287     
288                     //ensure type is on types string
289                     Assert.assertNotSame("Type not found on types string: "+type,Integer.valueOf(-1),Integer.valueOf(typeString.indexOf(type)));
290                     int index1=typeString.indexOf(type+";");
291                     int index2=typeString.indexOf(";"+type);
292                     Assert.assertTrue((index1!=-1)||(index2!=-1));
293                     
294                     //ensure condition exists
295                     String conditionString=this.emptyFaxClientSpi.getConfigurationValue("org.fax4j.spi.adapter.internal.spi.condition."+type);
296                     Assert.assertNotNull(conditionString);
297                     Assert.assertNotSame(Integer.valueOf(0),Integer.valueOf(conditionString.length()));
298                 }
299             }
300         }
301     }

311     {
312         boolean isCI = ("true".equals(System.getenv("CI")));
313         Assume.assumeTrue(TestUtil.isWindowsOS() && (!isCI));
314 
315         Properties configuration=new Properties();
316         configuration.setProperty("org.fax4j.proxy.enabled","false");
317         configuration.setProperty("org.fax4j.spi.adapter.internal.spi.types","windows");
318         
319         String condition=this.emptyFaxClientSpi.getConfigurationValue("org.fax4j.spi.adapter.internal.spi.condition.windows");
320         Assert.assertNotSame(Integer.valueOf(-1),Integer.valueOf(condition.indexOf("OS:windows")));
321         Assert.assertNotSame(Integer.valueOf(-1),Integer.valueOf(condition.indexOf("native-lib:winfax")));
322         if(!TestUtil.isWindowsOS())
323         {
324             condition=condition.replace("OS:windows","OS:"+System.getProperty("os.name"));
325             condition=condition.replace("native-lib:winfax","native-lib:jawt");
326             configuration.setProperty("org.fax4j.spi.adapter.internal.spi.condition.windows",condition);
327         }
328         
329         AdapterFaxClientSpi spi=(AdapterFaxClientSpi)FaxClientSpiFactory.createFaxClientSpi("adapter",configuration);
330         FaxClientSpi internalFaxClientSpi=spi.getInternalFaxClientSpi();
331         Assert.assertNotNull(internalFaxClientSpi);
332         Assert.assertEquals(WindowsFaxClientSpi.class,internalFaxClientSpi.getClass());
